<!DOCTYPE html>
<html>
<head>
  <style>
    body {
      background-color: white;
    }

    .dot {
      position: absolute;
      width: 10px;
      height: 10px;
      background-color: black;
      border-radius: 50%;
    }
  </style>
</head>
<body>
  <div class="dot"></div>

  <script>
    const dot = document.querySelector('.dot');

    // Set the initial position of the dot
    let x = 0;
    let y = 0;
    dot.style.left = x + "px";
    dot.style.top = y + "px";

    // Listen for keydown events
    document.addEventListener('keydown', (event) => {
      if (event.key === "ArrowUp") {
        // Move the dot up by 10 pixels
        y -= 10;
        dot.style.top = y + "px";
      } else if (event.key === "ArrowDown") {
        // Move the dot down by 10 pixels
        y += 10;
        dot.style.top = y + "px";
      } else if (event.key === "ArrowLeft") {
        // Move the dot left by 10 pixels
        x -= 10;
        dot.style.left = x + "px";
      } else if (event.key === "ArrowRight") {
        // Move the dot right by 10 pixels
        x += 10;
        dot.style.left = x + "px";
      }
    });
  </script>
</body>
</html>
